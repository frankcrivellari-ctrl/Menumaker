<script>
// Daten
const haupt = ["Hähnchenbrust", "Rinderschmorbraten", "Linsenhack-Bällchen (vegan)", "Gemüse-Lasagne", "Seelachsfilet"];
const saettigung = ["Kartoffeln", "Reis", "Pasta", "Couscous", "Süßkartoffelwürfel"];
const gemuese = ["Brokkoli & Karotten", "Ofengemüse", "Rotkohl", "Zucchini-Paprika"];
const saucen = ["Tomatensugo", "Jus dunkel", "Kräutersauce (vegan)", "Curry-Kokos"];

function pickRandom(arr){ return arr[Math.floor(Math.random()*arr.length)]; }

function generateClassic(){
  return {typ:"Klassisch", haupt:pickRandom(haupt), beilage:pickRandom(saettigung), gemuese:pickRandom(gemuese), sauce:pickRandom(saucen)};
}
function generateTwoComp(){
  return {typ:"Zwei-Komponenten", kombi: pickRandom(haupt)+" + "+pickRandom(saettigung)};
}
function generateOnePot(){
  return {typ:"One-Pot", gericht: pickRandom(haupt)+"-Eintopf mit "+pickRandom(gemuese)};
}

function render(results){
  const wrap=document.getElementById("results"); wrap.innerHTML="";
  results.forEach(r=>{
    const div=document.createElement("div"); div.className="card";
    div.innerHTML = `<span class="badge">${r.typ}</span><br>`+
      (r.haupt? `<b>Haupt:</b> ${r.haupt}<br>`:"")+
      (r.beilage? `<b>Beilage:</b> ${r.beilage}<br>`:"")+
      (r.gemuese? `<b>Gemüse:</b> ${r.gemuese}<br>`:"")+
      (r.sauce? `<b>Sauce:</b> ${r.sauce}<br>`:"")+
      (r.kombi? `<b>Kombi:</b> ${r.kombi}<br>`:"")+
      (r.gericht? `<b>Gericht:</b> ${r.gericht}<br>`:"");
    wrap.appendChild(div);
  });
}

document.getElementById("generate").addEventListener("click",()=>{
  const mode=document.querySelector('input[name="modus"]:checked').value;
  const n=parseInt(document.getElementById("count").value)||4;
  const out=[];
  for(let i=0;i<n;i++){
    if(mode==="classic") out.push(generateClassic());
    if(mode==="twocomp") out.push(generateTwoComp());
    if(mode==="onepot") out.push(generateOnePot());
  }
  render(out);
  document.getElementById("status").textContent=`${out.length} Vorschläge erzeugt`;
});
</script>
